\section{Modelo de alocação de salas}

\subsection{Notação}

\subsubsection{Conjuntos}

\begin{itemize}

\item $C$ - Conjunto de cursos. Os elementos desse conjunto são denotados por $c$.

\item $CC$ - Conjunto de cursos compatíveis.

\item $CP$ - Conjunto de campus. Os elementos desse conjunto são denotados por $cp$.

\item $D$ - Conjunto de disciplinas. Os elementos desse conjunto são denotados por $d$.

\item $D_{oft}$ - Conjunto de disciplinas de uma oferta $oft$. 
Os elementos desse conjunto são denotados por $d_{oft}$.

\item $D_{c}$ - Conjunto de disciplinas de um curso $c$.

\item $Oft$ - Conjunto de ofertas de cursos. Os elementos desse conjunto são 
denotados por $oft$.

\item $Oft_{d}$ - Conjunto de ofertas de uma disciplina $d$. 
Os elementos desse conjunto são denotados por $oft_{d}$.

\item $SCAP_{u}$ - Conjunto de salas da unidade $u$ classificadas de acordo com as suas 
capacidades. Os elementos desse conjunto são denotados por $tps$.

\item $S_{u}$ - Conjunto de salas da unidade $u$. Os elementos desse conjunto são 
denotados por $s$.

\item $U$ - Conjunto de unidades. Os elementos desse conjunto são denotados por $u$.

\item $SL_{bc}$ - Conjunto de semanas letivas (calendarios) presentes em um bloco curricular $bc$.

\end{itemize}

\subsubsection{Dados do Modelo}

\begin{itemize}

\item $Cap_{s,u}$ - Capacidade da sala $s$ da unidade $u$.

\item $C_{d}$ - Total de créditos da disciplina $d$.

\item $HTPS_{s}$ - Total de créditos disponíveis para a sala $s$ na semana. É calculado como sendo o máximo entre os totais de
créditos disponíveis de todas as semanas letivas na sala $s$ na semana.

\item $I_{d}$ - máximo de turmas que podem ser abertas da disciplina $d$.

\item $MinAlunos$ - mínimo de alunos para abertura de turmas.

\item $M$ - big $M$.

\item $O_{cp}$ - conjunto de ofertas de um campus $cp$.

\end{itemize}

\subsubsection{Variáveis}

\begin{itemize}

\item $x_{i,d,u,s}$ - número de créditos da turma $i$ da disciplina $d$ na unidade $u$ na sala $s$. 
\item $o_{i,d,u,s}$ - indica se a turma $i$ da disciplina $d$ foi alocada na unidade $u$ na sala $s$. 
\item $z_{i,d,cp}$ - indica se houve abertura da $i$-ésima turma da disciplina $d$ no campus $cp$. 
\item $a_{i,d,u,s,oft}$ - número de alunos da oferta $oft$ alocados na turma $i$ da disciplina $d$ na unidade $u$ na sala $s$. 
\item $b_{i,d,c}$ - indica se algum aluno do curso $c$ foi alocado para a $i$-ésima turma da disciplina $d$. 
\item $c_{i,d,u,s,oft}$ - indica se a turma $i$ da disciplina $d$ atendendo a oferta $oft$ foi alocada na unidade $u$ na sala $s$.
\item $fd_{d,oft}$ - variável de folga que indica a demanda não atendida da disciplina $d$ oferta $oft$.
\item $fc_{i,d,c,c',cp}$ - variável de folga para a restrição em que o compartilhamento de turmas
com alunos de cursos diferentes é proibido. 
\item $H$ - maior número de créditos alocados para uma sala na semana.
\item $fs_{d,s,oft}$ - variável de folga que indica o número (excedente a 1) de turmas de uma disciplina $d$
atendendo uma oferta $oft$ alocadas em uma mesma sala $s$.

\end{itemize}

\subsection{Formulação}

\subsubsection{Função Objetivo}

$$
\begin{array}{rll}
   \mbox{MIN} &
       1000 \cdot \sum\limits_{oft \in O} \sum\limits_{d \in D_{oft}} fd_{d,oft}
      \\
      &
       + 10 \cdot \sum\limits_{cp \in CP} 
			\sum\limits_{d \in D}\sum\limits_{i \in I_{d}} z_{i,d,cp}
      \\
      &
       + \sum\limits_{c \in C} \sum\limits_{d \in D_{c}}\sum\limits_{i \in I_{d}} b_{i,d,c}
      \\
      &
       + 100 \cdot \sum\limits_{i \in I_{d}}\sum\limits_{d \in D_{oft}}
      \sum\limits_{u \in U}\sum\limits_{s \in S_{u}} \sum\limits_{oft \in Oft} c_{i,d,u,s,oft}
      \\
      &
      + 100 \cdot \sum\limits_{i \in I_{d}}\sum\limits_{d \in D_{c} \cap D_{c'}}
      \sum\limits_{c \in C} \sum\limits_{c' \ne c \in C} fc_{i,d,c,c'}
      \\
      &     
      + 100 \cdot \sum\limits_{oft \in O} \sum\limits_{d \in D_{oft}}\sum\limits_{s \in S} fs_{d,s,oft}
      \\
      & 
       + H
\end{array}
$$



\subsubsection{Carga horária de todas as turmas de cada disciplina}

\begin{eqnarray}
\sum\limits_{u \in U}\sum\limits_{s \in S_{u}} x_{i,d,u,s}  =  C_{d} \cdot z_{i,d,cp} \nonumber \qquad 
\forall d \in D \quad
\forall i \in I_{d} \quad 
\forall cp \in CP 
\end{eqnarray}



\subsubsection{Ativação da variável $o$}

\begin{eqnarray}
C_{d} \cdot o_{i,d,u,s}  \geq  x_{i,d,u,s}  \nonumber \qquad 
\forall d \in D \quad
\forall i \in I_{d} \quad
\forall u \in U \quad
\forall s \in S_{u} \quad
\end{eqnarray}


\subsubsection{Evitar sobreposição de turmas da mesma disciplina}

\begin{eqnarray}
\sum\limits_{u \in U} \sum\limits_{s \in S_{u}} o_{i,d,u,s}  \leq  1  \nonumber \qquad 
\forall d \in D \quad
\forall i \in I_{d} \quad
\end{eqnarray}



\subsubsection{Número de créditos por sala na semana}

\begin{eqnarray}
\sum\limits_{d \in D}\sum\limits_{i \in I_{d}} x_{i,d,u,s}  \leq  HTPS_{s}  \nonumber \qquad 
\forall u \in U \quad
\forall s \in S_{u} \quad
\end{eqnarray}


\subsubsection{Evitar alocação de turmas da mesma disciplina em campus diferentes}

\begin{eqnarray}
\sum\limits_{cp \in CP} z_{i,d,cp}  \leq  1  \nonumber \qquad 
\forall d \in D \quad
\forall i \in I_{d}
\end{eqnarray}



\subsubsection{Capacidade alocada tem que permitir atender demanda da disciplina}

\begin{eqnarray}
\sum\limits_{u \in U} \sum\limits_{s \in S_{u}} \sum\limits_{i \in I_{d}} a_{i,d,u,s,oft} + fd_{d,oft} =  P_{d,oft}  \nonumber \qquad 
\forall oft \in O \quad
\forall d \in D_{oft}
\end{eqnarray}



\subsubsection{Se alocar uma sala para uma turma, tem que respeitar capacidade da sala}

\begin{eqnarray}
\sum\limits_{oft \in Oft_{d}} a_{i,d,u,s,oft} \leq Cap_{s,u} \cdot o_{i,d,u,s} 
\nonumber \qquad 
\forall u \in U \quad
\forall s \in S_{u} \quad
\forall d \in D \quad
\forall i \in I_{d} \quad
\end{eqnarray}


\subsubsection{Contabiliza se houve aluno de determinado curso alocado em uma turma (*)}

\begin{eqnarray}
\sum\limits_{oft \in Oft_{c}} \sum\limits_{s \in S_{u}} a_{i,d,s,oft} \leq M \cdot b_{i,d,c} \nonumber \qquad
\forall c \in C \quad
\forall d \in D \quad
\forall i \in I_{d}
\end{eqnarray}


\subsubsection{Não permitir que alunos de cursos diferentes compartilhem turmas (*)}

\begin{eqnarray} 
b_{i,d,c} + b_{i,d,c'} - fc_{i,d,c,c'} \leq 1 \nonumber \qquad 
\forall d \in D \quad
\forall i \in I_{d} \quad
\forall c,c' \in C\quad
\end{eqnarray}



\subsubsection{Não permitir que alunos de cursos incompatíveis compartilhem turmas (*)}

\begin{eqnarray} 
b_{i,d,c} + b_{i,d,c'}\leq 1 \nonumber \qquad 
\forall d \in D \quad
\forall i \in I_{d} \quad
\forall c,c' \notin CC \quad
\end{eqnarray}



\subsubsection{Associa variáveis o e z}

\begin{eqnarray}
\sum\limits_{u \in U} \sum\limits_{s \in S_{u}} o_{i,d,u,s} \leq M \cdot z_{i,d,cp} \nonumber \qquad 
\forall cp \in CP \quad
\forall d \in D \quad
\forall i \in I_{d}
\end{eqnarray}


\subsubsection{Limita a abertura de turmas}

\begin{eqnarray}
MinAlunos \cdot z_{i,d,cp} \leq \sum\limits_{u \in U} \sum\limits_{s \in S_{u}} \sum\limits_{oft \in Oft_{d}} a_{i,d,u,s,oft}  \nonumber \qquad 
\forall cp \in CP \quad
\forall d \in D \quad
\forall i \in I_{d}
\end{eqnarray}




\subsubsection{Abertura sequencial de turmas}

\begin{eqnarray}
\sum\limits_{u \in U} \sum\limits_{s \in S_{u}} o_{i,d,u,s} \geq \sum\limits_{u \in U} \sum\limits_{s \in S_{u}} o_{i',d,u,s} \nonumber \qquad 
\forall d \in D \quad
\forall i,i' \in I_{d} \quad i<i'
\end{eqnarray}



\subsubsection{Favorece turmas de mesma disciplina a serem abertas em salas diferentes}

\begin{eqnarray}
\sum\limits_{i \in I_{d}} c_{i,d,u,s,oft} + fs_{d,s,oft} \le 1 \nonumber \qquad 
\forall u \in U \quad
\forall s \in S_{u} \quad
\forall d \in D \quad
\forall oft \in Oft_{d} \quad
\end{eqnarray}



\subsubsection{Favorece a distribuição equilibrada das turmas entre as salas}

\begin{eqnarray}
H \ge \sum\limits_{i \in I_{d}} \sum\limits_{d \in D} x_{i,d,u,s} \nonumber \qquad 
\forall u \in U \quad
\forall s \in S_{u} \quad
\end{eqnarray}



\subsubsection{Ativação da variável c}

\begin{eqnarray}
a_{i,d,s,oft} \le M \cdot c_{i,d,u,s,oft} \nonumber \qquad 
\forall u \in U \quad
\forall s \in S_{u} \quad
\forall d \in D \quad
\forall i \in I_{d} \quad
\forall oft \in Oft_{d} \quad
\end{eqnarray}



\subsubsection{Fixa o Não-Compartilhamento}

\begin{eqnarray}
\sum\limits_{u \in U}\sum\limits_{s \in S_{u}} c_{i,d,u,s,oft1} +
\sum\limits_{u \in U}\sum\limits_{s \in S_{u}} c_{i,d,u,s,oft2} \le 1 \nonumber \qquad 
\forall i \in I_{d} \quad
\forall d \in D_{bc1} \cap D_{bc2} \quad
\forall bc1 \in BC_{oft1} \quad
\forall bc2 \in BC_{oft2} \quad
\text{sendo: } |SL_{bc1}| + |SL_{bc2}| > 2 \quad
\forall oft1 \in Oft \quad
\forall oft2 \in Oft \quad

\end{eqnarray}


\section{Módulo Tático}

\subsection{Notação}

\subsubsection{Conjuntos}

\begin{itemize}

\item $B$ - Conjunto de blocos curriculares. Os elementos desse conjunto são 
denotados por $bc$.

\item $B_{oft}$ - Conjunto de blocos curriculares de uma oferta $oft$.

\item $C$ - Conjunto de cursos. Os elementos desse conjunto são denotados por $c$.

\item $CC$ - Conjunto de cursos compatíveis.

\item $CP$ - Conjunto de campus. Os elementos desse conjunto são denotados por $cp$.

\item $D$ - Conjunto de disciplinas. Os elementos desse conjunto são denotados por $d$.

\item $D_{bc}$ - Conjunto de disciplinas que pertencem ao bloco curricular $bc$. 
Os elementos desse conjunto são denotados por $d_{bc}$.

\item $D_{oft}$ - Conjunto de disciplinas de uma oferta $oft$. 
Os elementos desse conjunto são denotados por $d_{oft}$.

\item $D_{incompat}$ - Conjunto de disciplinas incompatíveis.

\item $I_{bc}$ - Conjunto de turmas de um bloco curricular $bc$. 
Os elementos desse conjunto são denotados por $i_{bc}$.

\item $K_{d}$ - Conjunto de combinações possíveis de divisão de créditos de uma disciplina $d$. 
Os elementos desse conjunto são denotados por $k$.

\item $O$ - Conjunto de ofertas de cursos. Os elementos desse conjunto são 
denotados por $oft$.

\item $O_{d}$ - Conjunto de ofertas de uma disciplina $d$. 
Os elementos desse conjunto são denotados por $oft_{d}$.

\item $SCAP_{u}$ - Conjunto de salas da unidade $u$ classificadas de acordo com as suas 
capacidades. Os elementos desse conjunto são denotados por $tps$.

\item $S_{u}$ - Conjunto de salas da unidade $u$. Os elementos desse conjunto são 
denotados por $s$.

\item $T$ - Conjunto de dias letivos da semana. Os elementos desse conjunto 
são denotados por $t$.

\item $U$ - Conjunto de unidades. Os elementos desse conjunto são denotados por $u$.

\item $Kh_{s,t}$ - Conjunto de combinações possíveis de divisão de horários para uma sala $s$ em um dia $t$.
Os elementos desse conjunto são denotados por $kh_{s,t}$.

\item $Kh_{bc,t}$ - Conjunto de combinações possíveis de divisão de horários para um bloco curricular $bc$ em um dia $t$.
Os elementos desse conjunto são denotados por $kh_{bc,t}$.

\item $SL$ - Conjunto de semanas letivas. Os elementos desse conjunto são denotados por $sl$.


\end{itemize}

\subsubsection{Dados do Modelo}

\begin{itemize}

\item $A_{u,s}$ - capacidade da sala $s$ da unidade $u$.

\item $A_{u,tps}$ - capacidade total das salas de um conjunto de salas do tipo (capacidade) $tps$ da unidade $u$.

\item $C_{d}$ - Total de créditos da disciplina $d$.

\item $FC_{d,t}$ - número de créditos fixados para a disciplina $d$ no dia $t$.

\item $HTPS_{t,tps}$ - máximo de créditos permitidos por dia $t$ para o conjunto de salas do tipo (capacidade) $tps$.

\item $H_{t}$ - máximo de créditos permitidos por dia $t$.

\item $H_{bc,t}$ - máximo de créditos permitidos para a semana letiva do bloco $bc$ no dia $t$.

\item $I_{d}$ - máximo de turmas que podem ser abertas da disciplina $d$.

\item $M$ - big $M$.

\item $N_{d,k,t}$ - número de créditos determinados para a disciplina $d$ no dia $t$ na combinação de divisão de crédito $k$.

\item $O_{cp}$ - conjunto de ofertas de um campus $cp$.

\item $P_{d,c,cp}$ - demanda da disciplina $d$ no campus $cp$ para o curso $c$.

\item $P_{d,oft}$ - demanda da disciplina $d$ da oferta $oft$.

\item $Pmax_{d}$ - maior demanda da disciplina $d$.

\item $\alpha$ - peso associado a função objetivo.

\item $\beta$ - peso associado a função objetivo.

\item $\delta$ - peso associado a função objetivo.

\item $\gamma$ - peso associado a função objetivo.

\item $\lambda$ - peso associado a função objetivo.

\item $\omega$ - peso associado a função objetivo.

\item $\overline{H_{d}}$ - máximo de créditos diários da disciplina $d$.

\item $\psi$ - peso associado a função objetivo.

\item $\rho$ - peso associado a função objetivo.

\item $\tau$ - peso associado a função objetivo.

\item $\theta$ - peso associado a função objetivo.

\item $\underline{H_{d}}$ - mínimo de créditos diários da disciplina $d$.

\item $\xi$ - pesos associados a cada item da função objetivo.

\item $Q_{sl,kh,s,t}$ - número de horários da semana letiva $sl$ determinados para a sala $s$ no dia $t$ na combinação de divisão de horários $kh$.

\item $MaxCred_{oft,t}$ - máximo de créditos para a oferta $oft$ no dia $t$.

\item $Nh_{sl,kh,s,t}$ - número máximo de horários da semana letiva $sl$ determinados para a sala $s$ no dia $t$ na combinação de divisão de horários $kh \in Kh_{s,t}$.

\item $Nh_{sl,kh,bc,t}$ - número máximo de horários da semana letiva $sl$ determinados para o bloco curricular $bc$ no dia $t$ na combinação de divisão de horários $kh \in Kh_{bc,t}$.

\item $minAlunos$ - numero minimo de alunos necessários para abertura de uma turma. É um valor de parâmetro cadastrado pelo cliente.

\end{itemize}

\subsubsection{Variáveis}

\begin{itemize}

\item $\overline{h}_{bc,i}$ - máximo de créditos alocados na semana na $i$-ésima turma do bloco $bc$. 
\item $\underline{h}_{bc,i}$ - mínimo de créditos alocados na semana na $i$-ésima turma do bloco $bc$. 
\item $a_{i,d,oft}$ - número de alunos de uma oferta $oft$ alocados para a $i$-ésima turma da disciplina $d$. 
\item $b_{i,d,c,cp}$ - indica se algum aluno do curso $c$ foi alocado para a $i$-ésima turma da disciplina $d$ no campus $cp$. 
\item $bs_{i,d,c,c',cp}$ - variável de folga para a restrição em que o compartilhamento de turmas 
de alunos de cursos incompatíveis é proibido. 
\item $fc_{i,d,c,c',cp}$ - variável de folga para a restrição em que o compartilhamento de turmas 
de alunos de cursos diferentes (mesmo que compatíveis) é proibido. 
\item $c_{i,d,t}$ - indica se houve abertura de turma $i$ da disciplina $d$ em dias consecutivos. 
\item $fcm_{d,t}$ - variável de folga inferior para a restrição de fixação da distribuição de créditos por dia. 
\item $fcp_{d,t}$ - variável de folga superior para a restrição de fixação da distribuição de créditos por dia. 
\item $fd_{d,oft}$ - variável de folga para a restrição "Capacidade alocada tem que 
permitir atender demanda da disciplina". 
\item $fkm_{d,i,t}$ - variável de folga inferior para a restrição de combinação de divisão de créditos. 
\item $fkp_{d,i,t}$ - variável de folga superior para a restrição de combinação de divisão de créditos.
\item $fn_{bc,tps}$ - variável de folga para a restrição "Evitar alocação do mesmo 
bloco curricular em tipos de salas diferentes". 
\item $m_{d,i,k}$ - variável binária que indica se a combinação de divisão de créditos 
$k$ foi escolhida para a turma $i$ da disciplina $d$. 
\item $n_{bc,tps}$ - variável binária que indica se o bloco $bc$ foi alocado na sala $tps$. 
\item $o_{i,d,u,tps,t}$ - indica se a turma $i$ da disciplina $d$ foi alocada na unidade $u$ 
para alguma sala do tipo (capacidade) $tps$ no dia $t$. 
\item $r_{bc,t,cp}$ - indica se algum sub-bloco do bloco curricular $bc$ foi aberto no dia $t$ no campus $cp$. 
\item $v_{bc,t}$ - contabiliza a abertura do mesmo bloco curricular $bc$, no mesmo dia $t$, em campus distintos. 
\item $w_{bc,t,cp}$ - indica o número de sub-blocos abertos do bloco curricular $bc$ no dia $t$ no campus $cp$. 
\item $x_{i,d,u,tps,t}$ - número de créditos da turma $i$ da disciplina $d$ na unidade $u$ 
em salas do tipo (capacidade) $tps$ no dia $t$. 
\item $xm_{d,t}$ - máximo de créditos alocados para qualquer turma da disciplina $d$ no dia $t$. 
\item $y_{i,d,tps,u}$ - indica que a turma $i$ da disciplina $d$ foi alocada em alguma sala do tipo $tps$ da unidade $u$. 
\item $z_{i,d,cp}$ - indica se houve abertura da $i$-ésima turma da disciplina $d$ no campus $cp$. 
\item $zc_{d,t}$ - indica se houve abertura da disciplina $d$ no dia $t$.
\item $e_{i,d,oft}$ - indica se houve alunos de uma oferta $oft$ alocados para a $i$-ésima turma da disciplina $d$. 
\item $of_{i,d,oft1,oft2}$ - indica se houve tanto alunos da oferta $oft1$ quanto da oferta $oft2$ alocados para a $i$-ésima turma da disciplina $d$.
\item $q_{i,d,oft,u,tps,t}$ - número de créditos alocados para a oferta $oft$, turma $i$ da disciplina $d$ na unidade $u$ 
em salas do tipo (capacidade) $tps$ no dia $t$. 
\item $p_{i,d,oft1,oft2,u,tps,t}$ - número de créditos alocados tanto para a oferta $oft1$ quanto para a $oft2$, turma $i$ da disciplina $d$ na unidade $u$ em salas do tipo (capacidade) $tps$ no dia $t$. 
\item $g_{d,oft,t}$ - número mínimo de horários que a disciplina $d$ oferta $oft$ ocupa no dia t. É relevante quando a disciplina de uma oferta possui mais de 1 turma.
\item $cs_{s,t,k}$ - variável binária que indica se a combinação de divisão de horários $k$ foi escolhida para a sala $s$ no dia $t$.
\item $cbc_{bc,t,k}$ - variável binária que indica se a combinação de divisão de horários $k$ foi escolhida para o bloco curricular $bc$ no dia $t$.

\end{itemize}

\subsection{Formulação}

\subsubsection{Função Objetivo}

$$
\begin{array}{rll}
   \mbox{MIN} &
      \lambda \cdot \sum\limits_{bc \in B} 
			\sum\limits_{i \in I_{d}, d \in D_{bc}} \left( \overline{h}_{bc,i} - \underline{h}_{bc,i} \right)
      \\
      &
       + \delta \cdot \sum\limits_{d \in D} 
			\sum\limits_{i \in I_{d}} \sum\limits_{t \in T-{1}} c_{i,d,t}
      \\
      &
       + \xi \cdot \sum\limits_{d \in D} \sum\limits_{t \in T} fcm_{d,t}
      \\
      &
       + \xi \cdot \sum\limits_{d \in D} \sum\limits_{t \in T} fcp_{d,t}
      \\
      &
       + \omega \cdot \sum\limits_{oft \in O} \sum\limits_{d \in D_{oft}} fd_{d,oft}
      \\
      &
       + \psi \cdot \sum\limits_{d \in D} 
			\sum\limits_{t \in T} \sum\limits_{i \in I_{d}} fk_{d,i,k}
      \\
      &
       + \tau \cdot \sum\limits_{bc \in B} \sum\limits_{tps \in SCAP_{u}} fn_{bc,tps}
      \\
      &
       + \beta \cdot \sum\limits_{bc \in B} \sum\limits_{t \in T} v_{b,t}
      \\
      &
       + \rho \cdot \sum\limits_{bc \in B}\sum\limits_{t \in T} (\sum\limits_{cp \in CP} w_{bc,t,cp})
      \\
      &
       + \theta \cdot \sum\limits_{u \in U}\sum\limits_{tps \in SCAP_{u}} 
			\sum\limits_{d \in D}\sum\limits_{t \in T}
			\sum\limits_{i \in I_{d}} x_{i,d,u,tps,t}
      \\
      &
       + \alpha \cdot \sum\limits_{d \in D} \sum\limits_{cp \in CP}\sum\limits_{i \in I_{d}}z_{i,d,cp}
				+ \gamma \cdot \sum\limits_{d \in D} \sum\limits_{cp \in CP}\sum\limits_{i \in I_{d}}
				\left(\frac{Pmax_{d} - \sum\limits_{c \in C}P_{d,c,cp}}{Pmax_{d}} \right) \cdot z_{i,d,cp}
      \\
      &
       + 100 \cdot \sum\limits_{bc \in B}\sum\limits_{t \in T}\sum\limits_{c,c' \notin CC}\sum\limits_{cp \in CP} bs_{i,d,c,c',cp}
      \\
      &
       + 100 \cdot \sum\limits_{d \in D} \sum\limits_{i \in I_{d}} \sum\limits_{c,c' \in CC} \sum\limits_{cp \in CP} fc_{i,d,c,c',cp}
\end{array}
